import java.io.File

fun String.isFilexists(): Boolean {
    val dir = File(this)
    return dir.exists()
}
//这个 this 关键字在扩展函数内部对应到接收者对象（传过来的在点符号前的对象）这个应该不难理解吧。
fun String.toast(context: Context):String {
    Toast.makeText(context,this,Toast.LENGTH_LONG).show()
    return this;
